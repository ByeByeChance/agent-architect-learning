### Task 5: Agent 实践 — Weather MCP Server

**Files:** Create: `servers/weather-server/index.ts`

- [ ] **Step 1: 创建工具 + 资源**

```typescript
import { McpServer } from "@modelcontextprotocol/sdk/server/mcp.js";
import { StdioServerTransport } from "@modelcontextprotocol/sdk/server/stdio.js";
import { z } from "zod";

const server = new McpServer({ name: "weather-server", version: "1.0.0" });

// Tool: 获取天气
server.tool("get_weather", "获取指定城市天气", {
  city: z.string().describe("城市名称"),
}, async ({ city }) => {
  const data: Record<string, string> = {
    北京: "晴 25°C", 上海: "多云 28°C", 深圳: "阵雨 30°C",
  };
  return { content: [{ type: "text", text: data[city] || `${city}: 数据暂无` }] };
});

// Resource: 暴露城市列表供 agent 读取
server.resource("cities", "weather://cities", async () => ({
  contents: [{ uri: "weather://cities", text: "北京, 上海, 深圳", mimeType: "text/plain" }],
}));

const transport = new StdioServerTransport();
await server.connect(transport);
```

### Task 6: Agent 实践 — File Search MCP Server

**Files:** Create: `servers/file-search-server/index.ts`

- [ ] **Step 1: 实现文件搜索工具**

```typescript
import { McpServer } from "@modelcontextprotocol/sdk/server/mcp.js";
import { StdioServerTransport } from "@modelcontextprotocol/sdk/server/stdio.js";
import { z } from "zod";
import fs from "fs";
import path from "path";

const server = new McpServer({ name: "file-search-server", version: "1.0.0" });

const SEARCH_ROOT = process.env.SEARCH_ROOT || process.cwd();

server.tool("search_files", "搜索文件名", {
  pattern: z.string().describe("搜索关键词"),
  maxResults: z.number().default(20).describe("最大结果数"),
}, async ({ pattern, maxResults }) => {
  const results: string[] = [];
  function walk(dir: string) {
    if (results.length >= maxResults) return;
    try {
      for (const entry of fs.readdirSync(dir)) {
        const full = path.join(dir, entry);
        if (entry.includes(pattern)) results.push(full);
        if (fs.statSync(full).isDirectory()) walk(full);
      }
    } catch {}
  }
  walk(SEARCH_ROOT);
  return { content: [{ type: "text", text: results.join("\n") || "无结果" }] };
});

// Resource: 暴露搜索根目录
server.resource("search-root", "search://root", async () => ({
  contents: [{ uri: "search://root", text: SEARCH_ROOT, mimeType: "text/plain" }],
}));

const transport = new StdioServerTransport();
await server.connect(transport);
```

### Task 11: 源码溯源 — MCP vs OpenAI Function Calling

**Files:** Create: `phase-2-mcp/research/mcp-vs-openai-tools.md`

- [ ] **Step 1: 对照分析**

```markdown
# MCP vs OpenAI Function Calling

| 维度 | MCP | OpenAI Tools |
|---|---|---|
| 协议 | JSON-RPC over Stdio/SSE | HTTP REST |
| 工具注册 | server.tool() | tools 参数 |
| 资源暴露 | server.resource() | 无 |
| 发现机制 | tools/list 方法 | 无（需预先知道） |
| 适用场景 | IDE agent、本地工具 | Web API 调用 |

## 什么时候用哪个？
- MCP：需要 agent 主动发现工具、需要暴露数据资源
- OpenAI Tools：简单的 API 调用、不需要发现机制
```
